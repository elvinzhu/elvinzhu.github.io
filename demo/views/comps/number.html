@param canEdit{bool}: 是否允许用户编辑
@param el{element}: dom 对象， 为避免用户感知到元素从无到有， dom结构需提前存在
@param min{int}: 允许的最小值
@param min{max}: 允许的最大值
@param onchange{function}: 新值生效之前触发，返回false可取消新值
@param change{function}: 新值生效之后触发

this.quantity = new UINumber({
    canEdit: true,
    el: this.$('#buyCount'),
    min: 1,
    max: 1,
    onchange: function( data ){

    },
    change: function( data ){

        console.log(data);
    }
})
